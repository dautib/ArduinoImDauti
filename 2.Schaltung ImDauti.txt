#include <LiquidCrystal.h>
#include <Keypad.h>
 
LiquidCrystal lcd(13, 12, 11, 10, 9, 8);  
// Initialisierung Variabeln
long erste = 0;
long zweite = 0;
double total = 0;
int posit = 0 ;
char zeichen;
const byte REIHEN = 4;
const byte SPALTEN = 4;

// Buttons zu Zeichen zuteilen
char zahlen[REIHEN][SPALTEN] = { 
  {'1','2','3','/'},
  {'4','5','6','*'},
  {'7','8','9','-'},
  {'C','0','=','+'}
};
byte reihenPins[REIHEN] = {7 ,6 ,5 ,4}; 
byte spaltenPins[SPALTEN] = {3, 2, 1, 0}; 

Keypad eigenesZeichenPad = Keypad( makeKeymap(zahlen), reihenPins, spaltenPins, REIHEN, SPALTEN); 

// Methode fÃ¼r das Setup 
void setup(){
  lcd.begin(16,2);
  lcd.setCursor(7,0);
  lcd.print("Im");
  lcd.setCursor(6,1);
  lcd.print("Dauti");
  delay(3000);
  lcd.begin(16,2);
  lcd.setCursor(3,0);
  lcd.print("Calculator");
  lcd.setCursor(5,1);
  lcd.print("Arduino");
  delay(2000);
  lcd.clear(); 
}
void loop()
{
  zeichen = eigenesZeichenPad.getKey();
  switch(zeichen) 
  {
// Bei Auswahl der Zahlen
  case '0' ... '9': 
    lcd.setCursor(0,0);
    erste = erste * 10 + (zeichen - '0');
    lcd.print(erste);
    posit++;
    break;
//Bei Addition
  case '+':
    erste = (total != 0 ? total : erste);
    lcd.setCursor(posit,0);
    lcd.print("+");
    posit++;
    zweite = ZweiteNummer(); 
    total = erste + zweite;
    lcd.setCursor(1,1);
    lcd.print(total);
    erste = 0, zweite = 0; 
    posit=0;
    break;
// Bei Subtraktion
  case '-':
    erste = (total != 0 ? total : erste);
    lcd.setCursor(posit,0);
    lcd.print("-");
    posit++;
    zweite = ZweiteNummer();
    total = erste - zweite;
    lcd.setCursor(1,1);
    lcd.print(total);
    erste = 0, zweite = 0;
    posit=0;
    break;
// Bei Multiplikation
  case '*':
    erste = (total != 0 ? total : erste);
    lcd.setCursor(posit,0);
    lcd.print("*");
    posit++;
    zweite = ZweiteNummer();
    total = erste * zweite;
    lcd.setCursor(1,1);
    lcd.print(total);
    erste = 0, zweite = 0;
    posit=0;
    break;
// Bei Division
  case '/':
    erste = (total != 0 ? total : erste);
    lcd.setCursor(posit,0);
    lcd.print("/");
    posit++;
    zweite = ZweiteNummer();
    lcd.setCursor(1,1);

    zweite == 0 ? lcd.print("Error") : total = (float)erste / (float)zweite;

    lcd.print(total);
    erste = 0, zweite = 0;
    posit=0;
    break;
// wenn alles resettet
  case 'C':
    total = 0;
    erste = 0;
    zweite = 0;
    posit = 0;
    lcd.clear();
    break;
  }
}
// Eingabe der 2. Nummer
long ZweiteNummer()
{
  while( 1 )
  {
    zeichen = eigenesZeichenPad.getKey();
    if(zeichen >= '0' && zeichen <= '9')
    {
      zweite = zweite * 10 + (zeichen - '0');
      lcd.setCursor(posit,0);
      lcd.print(zweite);
    }
    if(zeichen == 'C') {
      total = 0;
      erste = 0;
      zweite = 0;
      posit = 0;
      lcd.clear();
      break;
    }

    if(zeichen == '='){
      lcd.setCursor(0,1);
          lcd.print("=");
      
      posit = total;
      lcd.clear();
   lcd.setCursor(0,1);
          lcd.print("=");
      break;
    }
  }
 return zweite;
}
